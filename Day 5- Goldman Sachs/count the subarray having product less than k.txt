Problem Name: Count the subarray having product less than K
Problem Link: https://practice.geeksforgeeks.org/problems/count-the-subarrays-having-product-less-than-k1708/1/#

Code:
#include <bits/stdc++.h>
using namespace std;
class Solution{
  public:
    int countSubArrayProductLessThanK(const vector<int>& a, int n, long long k) {
        
        int i=0,ans=0;
        long long res=1;
        for(int j=0;j<n;j++)
        {
            res=res*a[j];
           
            if(res<k)
            ans+=j-i+1;
            else
            {
                while(i<=j && res>=k)
                { res/=a[i];
                i++;}
                ans+=j-i+1;
            }
            
        }
        return ans;
        
    }
};

int main() {
    int t;

    while (t--) {
        int n, i;

        cin >> n >> k;
        vector<int> arr(n);
        for (i = 0; i < n; i++) cin >> arr[i];
        Solution obj;
        cout << obj.countSubArrayProductLessThanK(arr, n, k) << endl;
    }

}
